% Space is not at premium since so 12pt is a good choice.
% There is no particular need for double sided document if it is not printed.
% Other options also make sense when considering the use case.
\documentclass[12pt, a4paper, final, onecolumn, oneside]{article}

% Big margins would waste space which may be used to neatly display figures.
% We could set them manually, but it's much more convenient with geometry package.
% Showframe may be used to see the margins while setting them.
%\usepackage{showframe}
\usepackage[top=1in, bottom=1in, left=1in, right=1in]{geometry}

% We want to have good-looking clickable URLs to external docs,
% without having the red default borders around them.
\usepackage[pdfborder={0 0 0}]{hyperref}

% Not a necessity but some think it looks better when
% indenting the first paragraph as well.
\usepackage{indentfirst}

% Used to neatly manage pictures, not much to say about this one.
\usepackage{graphicx}

% We want the figures of each task to appear strictly in that task (section).
\usepackage[section]{placeins}

% For better looking captioning of figures.
\usepackage{caption}
